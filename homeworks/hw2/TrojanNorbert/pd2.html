library(ggplot2)
library(dplyr)
library(scales)
library(tidyverse)
porzdkowa<-c(1,2,3,4)
procenty<-c(-0.002, 0.055, -0.001, 0.031)
podpis<-c('Month to month ','Year to year ','Month to month','Year to year')
dane<- data.frame(porzdkowa=porzdkowa,procenty=procenty,podpis=podpis)


opisy<-c('-0.2%','5.5%','-0.1%','3.1%')

dane<- dane %>%
   mutate(opisy=opisy)

 
poprawiony_wykres<-ggplot(dane,aes(x=reorder(podpis,porzdkowa),y=procenty,label =opisy))+
  geom_col()+ 
  scale_y_continuous(labels=scales::percent)+
  geom_text(nudge_y = .004)+
  geom_vline(xintercept = 2.5)+
  labs(title = "Procentowa zmianna cen nieruchomoi",
       x = "Czy zmina jest liczona z miesica na miesi, czy z roku na rok?",
       y = element_blank())+
  annotate("text", x=1.5, y=0.06, label= "Obecne zmiany",size=5)+
  annotate("text", x=3.5, y=0.06, label= "Przewidywane zmiany",size=5)
